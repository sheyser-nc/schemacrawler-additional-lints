33  public class JSonUtilsTest {
34      
35      public JSonUtilsTest(){
36          JSonUtils test = new JSonUtils();
37          Assert.assertTrue(true);
38      }
39      @Test
40      public void testUtils_success() throws Exception{
41          
42          String data = "{\"menu\": {\"id\": \"file\", \"value\": \"File\", \"popup\": {\"menuitem\": [{\"value\": \"New\", "
43                  + "\"onclick\": \"CreateNewDoc()\"},{\"value\": \"Open\", \"onclick\": \"OpenDoc()\"},"
44                  + " {\"value\": \"Close\", \"onclick\": \"CloseDoc()\"}]}}}";
45          
46          Assert.assertTrue(JSonUtils.isJsonContent(data));
47          
48      }
49      
50      @Test
51      public void testUtils_fails() throws Exception{
52          
53          String data = "Lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do eiusmod tempor incididunt ut labore et dolore magna aliqua."
54                  + " Ut enim ad minim veniam, quis nostrud exercitation ullamco laboris nisi ut aliquip ex ea commodo consequat. Duis aute irure "
55                  + "dolor in reprehenderit in voluptate velit esse cillum dolore eu fugiat nulla pariatur. Excepteur sint occaecat cupidatat non "
56                  + "proident, sunt in culpa qui officia deserunt mollit anim id est laborum.";
57          
58          Assert.assertFalse(JSonUtils.isJsonContent(data));
59          
60          Assert.assertFalse(JSonUtils.isJsonContent(null));
61          
62      }
63  }
