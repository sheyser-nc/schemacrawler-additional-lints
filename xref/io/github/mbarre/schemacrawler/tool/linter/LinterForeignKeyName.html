25  import schemacrawler.schema.*;
26  import schemacrawler.schemacrawler.Config;
27  import schemacrawler.schemacrawler.SchemaCrawlerException;
28  import schemacrawler.tools.lint.BaseLinter;
29  import schemacrawler.tools.lint.LintSeverity;
30  
31  import java.sql.Connection;
32  import java.util.logging.Level;
33  import java.util.logging.Logger;
34  
35  import static java.util.Objects.requireNonNull;
36  
37  /**
38   *  Check that foreign key name has right prefix or suffix name
39   * Created by barmi83 on 12/09/16.
40   */
41  
42  public class LinterForeignKeyName  extends BaseLinter {
43  
44      private static final Logger LOGGER = Logger.getLogger(LinterForeignKeyName.class.getName());
45  
46      public static final String PREFIX_CONFIG_PARAM = "foreignKeyPrefix";
47      public static final String SUFFIX_CONFIG_PARAM = "foreignKeySuffix";
48  
49      private String prefix;
50      private String suffix;
51  
52      public LinterForeignKeyName() {
53          prefix = "";
54          suffix = "";
55          setSeverity(LintSeverity.medium);
56      }
57  
58      @Override
59      public String getSummary() {
60          return "FK name does not contain required prefix or suffix.";
61      }
62  
63      /**
64       * Get the description
65       * @return the description
66       */
67      @Override
68      public String getDescription() {
69          return "Foreign key data type does not match Primary key";
70      }
71  
72      @Override
73      public void configure(Config config) {
74          prefix = config.getStringValue(PREFIX_CONFIG_PARAM, "");
75          suffix = config.getStringValue(SUFFIX_CONFIG_PARAM, "");
76      }
77  
78      @Override
79      protected void lint(Table table, Connection connection) throws SchemaCrawlerException {
80  
81          LOGGER.log(Level.CONFIG, "<"+PREFIX_CONFIG_PARAM+"> parameter set to {0}%", prefix);
82          LOGGER.log(Level.CONFIG, "<"+SUFFIX_CONFIG_PARAM+"> parameter set to {0}%", suffix);
83  
84          requireNonNull(table, "No table provided");
85          findMismatchedForeignKeys(table);
86  
87      }
88  
89      private void findMismatchedForeignKeys(final Table table){
90  
91          if(prefix.isEmpty() && suffix.isEmpty())
92              return;
93  
94          if (table != null && !(table instanceof View))
95          {
96              for (final ForeignKey foreignKey: table.getImportedForeignKeys())
97              {
98                  for (final ForeignKeyColumnReference columnReference: foreignKey)
99                  {
100                     final Column fkColumn = columnReference.getForeignKeyColumn();
101                     if (fkColumn != null && !fkColumn.getName().startsWith(prefix))
102                     {
103                         addTableLint(table, "Foreign key does not start with prefix \"" + prefix +"\"", foreignKey);
104 
105                     }
106                     if (fkColumn != null && !fkColumn.getName().endsWith(suffix))
107                     {
108                         addTableLint(table, "Foreign key does not ends with suffix \"" + suffix +"\"", foreignKey);
109                     }
110                 }
111             }
112         }
113     }
114 
115 }
